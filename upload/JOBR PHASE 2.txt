```prisma
model Organization {
  id             String   @id @default(uuid())
  orgName        String   @map("org_name")

  headquarters   String?  @map("headquarters")
  socialLinks    Json?    @map("social_links")
  
  seoTitle       String?  @map("seo_title")

  deletedAt      DateTime? @map("deleted_at")
  
  isVerified     Boolean  @default(false) @map("is_verified")
  isActive       Boolean  @default(true)  @map("is_active")

  jobs                  Job[]

  @@index([orgIndustry, orgType])
```

## 5. Implementation Guide (Service Layer & Search)

### ✅ Populating `searchText` (Service Layer)
```ts
// services/job.service.ts
export async function createJob(data: CreateJobInput) {
  const org = await prisma.organization.findUnique({
    where: { id: data.organizationId },
    select: { orgName: true },
  });

  const subcategory = await prisma.jobSubcategory.findUnique({
    where: { id: data.subcategoryId },
    select: { categoryId: true },
  });

  const searchText = [
    data.title,
    data.description,
    org?.orgName || '',
    data.locationCity,
    data.locationCounty,
  ].filter(Boolean).join(' ');

  return prisma.job.create({
    data: {
      ...data,
      categoryId: subcategory?.categoryId,
      searchText,
    },
  });
}
```

### ✅ MySQL Full-Text Search
```ts
export async function searchJobs(query: string) {
  return prisma.$queryRaw`
    SELECT j.*, o."orgName", o."orgLogoUrl"
    FROM "jobs" j
    LEFT JOIN "organizations" o ON j."organizationId" = o.id
    WHERE 
      j."status" = 'ACTIVE' 
      AND j."deletedAt" IS NULL
      AND MATCH(j."search_text") AGAINST(${query} IN NATURAL LANGUAGE MODE)
    ORDER BY MATCH(j."search_text") AGAINST(${query} IN NATURAL LANGUAGE MODE) DESC
    LIMIT 50
  `;
}
```

### ✅ SEO Metadata (Next.js 15)
```tsx
export async function generateMetadata({ params }: Props) {
  const job = await getJob(params.slug);
  const title = job.seoTitle || `${job.title} - ${job.organization.orgName}`;
  const description = job.seoDescription || job.description.slice(0, 160);

  return {
    title,
    description,
    alternates: { canonical: `/jobs/${job.slug}` },
    robots: job.status === "ACTIVE" ? { index: true } : { index: false },
  };
}
```

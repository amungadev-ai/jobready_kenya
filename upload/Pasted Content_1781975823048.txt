    document.addEventListener('DOMContentLoaded', function() {
      // ----- TABS -----
      const tabs = document.querySelectorAll('.tab-btn');
      const contents = {
        entry: document.getElementById('entry'),
        internships: document.getElementById('internships'),
        scholarships: document.getElementById('scholarships')
      };

      tabs.forEach(tab => {
        tab.addEventListener('click', function() {
          tabs.forEach(t => {
            t.classList.remove('active');
            t.classList.add('inactive');
            const count = t.querySelector('.tab-count');
            if (count) count.className = 'tab-count text-gray-400';
          });
          this.classList.remove('inactive');
          this.classList.add('active');
          const count = this.querySelector('.tab-count');
          if (count) count.className = 'tab-count text-emerald-200';
          Object.values(contents).forEach(content => content.classList.remove('active'));
          const tabId = this.getAttribute('data-tab');
          if (contents[tabId]) contents[tabId].classList.add('active');
        });
      });

      // ----- HELPER FUNCTION FOR SCROLL CONTAINERS -----
      function setupScroll(containerId, leftBtnId, rightBtnId, scrollAmount = 300) {
        const container = document.getElementById(containerId);
        const leftBtn = document.getElementById(leftBtnId);
        const rightBtn = document.getElementById(rightBtnId);

        if (!container || !leftBtn || !rightBtn) return;

        function updateButtons() {
          const maxScroll = container.scrollWidth - container.clientWidth;
          if (container.scrollLeft <= 0) {
            leftBtn.classList.add('disabled');
          } else {
            leftBtn.classList.remove('disabled');
          }
          if (container.scrollLeft >= maxScroll - 1) {
            rightBtn.classList.add('disabled');
          } else {
            rightBtn.classList.remove('disabled');
          }
        }

        leftBtn.addEventListener('click', () => {
          container.scrollBy({ left: -scrollAmount, behavior: 'smooth' });
        });
        rightBtn.addEventListener('click', () => {
          container.scrollBy({ left: scrollAmount, behavior: 'smooth' });
        });
        container.addEventListener('scroll', updateButtons);
        window.addEventListener('resize', updateButtons);
        updateButtons();
      }

      // Setup Categories Scroll
      setupScroll('categories-scroll', 'scroll-left-cat', 'scroll-right-cat', 300);
    });
  </script>
